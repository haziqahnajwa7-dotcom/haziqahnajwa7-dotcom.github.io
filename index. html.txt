    <script>
        // --- Core Mathematical Formula Simulation State ---
        let L = 25.0;   // Maximum Bunch Weight Carrying Capacity (kg)
        let k = 0.40;   // Growth rate coefficient (influence curve shape)
        let t0 = 11.0;  // Inflection point midpoint (weeks)
        let selectedWeek = 11.0; // Currently inspected timeline index

        let chart = null;

        // --- Math Functions ---
        // f(t) = L / (1 + exp(-k * (t - t0)))
        function f(t) {
            return L / (1 + Math.exp(-k * (t - t0)));
        }

        // f'(t) = (k * L * exp(-k * (t - t0))) / (1 + exp(-k * (t - t0)))^2
        function fPrime(t) {
            const expTerm = Math.exp(-k * (t - t0));
            return (k * L * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // --- Layout Data Generators ---
        function generateData() {
            const data = {
                labels: [],
                weight: [],
                rate: []
            };
            // Generate steps every 0.2 weeks for smooth line resolution
            for (let t = 0; t <= 22; t = parseFloat((t + 0.2).toFixed(1))) {
                data.labels.push(t);
                data.weight.push(f(t));
                data.rate.push(fPrime(t));
            }
            return data;
        }

        // --- Layman Agronomic Explanations Generator ---
        function getAgronomicAdvice(t, weight, rate) {
            let title = "";
            let desc = "";
            let phase = "";
            let tip = "";

            if (t < 5) {
                title = "Lag Phase: Pollination & Early Inflorescence";
                desc = `During weeks 0 to 5, the oil palm fruit bunch undergoes cell division immediately following anthesis. Physical weight gains are minimal (${weight.toFixed(2)} kg), and the rate of weight accumulation is low at ${rate.toFixed(2)} kg/week. Energy is allocated to structural fruitlet establishment.`;
                phase = "Establishment";
                tip = "Apply Early Nitrogen (N)";
            } else if (t >= 5 && t < 9) {
                title = "Early Growth Acceleration Phase";
                desc = `The fruit bunch starts rapidly growing in volume. Cells are expanding. The rate of change is rising to ${rate.toFixed(2)} kg/week. Potassium intake becomes highly crucial here to sustain cellular turgor pressure.`;
                phase = "Cell Expansion";
                tip = "Irrigation Control";
            } else if (t >= 9 && t < 13) {
                title = "Inflection Point (Peak Growth Rate)";
                desc = `Around week ${t0}, the derivative peaks at its maximum of ${rate.toFixed(2)} kg/week! This is the peak linear growth phase. Any fertilizer deficiencies experienced now will permanently stunt the ultimate harvest weight of the bunch.`;
                phase = "Linear Surge";
                tip = "Apply High Potassium (K+)";
            } else if (t >= 13 && t < 17) {
                title = "Growth Deceleration & Mesocarp Formation";
                desc = `The derivative drops to ${rate.toFixed(2)} kg/week, which means physical weight increase is slowing down, though bunch weight has reached a heavy ${weight.toFixed(2)} kg. The palm bunch is transitionally changing from size growth to inner lipid accumulation.`;
                phase = "Slowing Expansion";
                tip = "Monitor Bunches";
            } else {
                title = "Plateau Phase: Lipids Synthesis & Harvest Ripening";
                desc = `At week ${t.toFixed(1)}, the growth rate approaches 0 (${rate.toFixed(2)} kg/week). The derivative leveling off shows the bunch has stopped expanding physically. However, internal fatty acids are converting rapidly to palm oil. The ideal Malaysian harvest begins as the derivative goes flat and the first loose fruits drop!`;
                phase = "Peak Ripening";
                tip = "Harvest Instantly";
            }

            return { title, desc, phase, tip };
        }

        // --- Update UI with Current Mathematical Values ---
        function updateUI() {
            const wt = f(selectedWeek);
            const dw = fPrime(selectedWeek);

            // Update badge & outputs
            document.getElementById('panel-week-badge').innerText = `Week ${selectedWeek.toFixed(1)}`;
            document.getElementById('output-wt').innerText = `${wt.toFixed(2)} kg`;
            document.getElementById('output-dw').innerText = `${dw.toFixed(2)} kg/wk`;
            
            // Formula display variables updating
            document.getElementById('formula-numerator').innerText = L.toFixed(1);

            // Detailed mathematical text
            document.getElementById('panel-math-desc').innerHTML = 
                `At week <span class="text-white font-semibold">${selectedWeek.toFixed(1)}</span> post-anthesis:<br>` +
                `<span class="text-emerald-400 font-semibold">W(${selectedWeek.toFixed(1)})</span> = <span class="text-white">${wt.toFixed(2)} kg</span><br>` +
                `<span class="text-lime-400 font-semibold">W'(${selectedWeek.toFixed(1)})</span> = <span class="text-white">${dw.toFixed(3)} kg/week</span><br>` +
                `The derivative value indicates that the bunch weight is increasing by exactly <span class="text-white">${dw.toFixed(2)} kilograms</span> per week at this instantaneous point in time.`;

            // Layman Agronomy
            const advice = getAgronomicAdvice(selectedWeek, wt, dw);
            document.getElementById('layman-title').innerText = advice.title;
            document.getElementById('layman-desc').innerText = advice.desc;
            document.getElementById('panel-phase-name').innerText = advice.phase;
            document.getElementById('panel-agri-tip').innerText = advice.tip;

            // Update slider position (in case updated via chart clicks)
            document.getElementById('scrubber-t').value = selectedWeek;

            // Draw a vertical marker representing the current week selection on the chart
            if (chart) {
                chart.update('none'); // Update without full re-animation to stay smooth
            }
        }

        // --- STREAMING_CHUNK: Setup and configure the Chart.js visualization canvas ---
        function initChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            const dataSets = generateData();

            chart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: dataSets.labels,
                    datasets: [
                        {
                            label: 'Bunch Weight W(t) (kg)',
                            data: dataSets.weight,
                            borderColor: '#34d399',
                            backgroundColor: 'rgba(52, 211, 153, 0.05)',
                            yAxisID: 'yWeight',
                            borderWidth: 3.5,
                            pointRadius: 0,
                            pointHoverRadius: 6,
                            tension: 0.1,
                            fill: true
                        },
                        {
                            label: "Growth Rate W'(t) (kg/wk)",
                            data: dataSets.rate,
                            borderColor: '#a3e635',
                            backgroundColor: 'rgba(163, 230, 53, 0.05)',
                            yAxisID: 'yRate',
                            borderWidth: 3,
                            borderDash: [5, 5],
                            pointRadius: 0,
                            pointHoverRadius: 6,
                            tension: 0.1,
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    onClick: (e) => {
                        const points = chart.getElementsAtEventForMode(e, 'nearest', { intersect: false }, true);
                        if (points.length) {
                            const index = points[0].index;
                            selectedWeek = parseFloat(chart.data.labels[index]);
                            updateUI();
                        }
                    },
                    plugins: {
                        legend: {
                            display: false // We use our own customized header legends
                        },
                        tooltip: {
                            mode: 'index',
                            intersect: false,
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                        label += context.parsed.y.toFixed(2);
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    scales: {
                        x: {
                            title: {
                                display: true,
                                text: 'Time Post-Anthesis (Weeks)',
                                color: '#9ca3af',
                                font: {
                                    weight: 'bold'
                                }
                            },
                            grid: {
                                color: 'rgba(255, 255, 255, 0.05)'
                            },
                            ticks: {
                                color: '#9ca3af'
                            }
                        },
                        yWeight: {
                            type: 'linear',
                            position: 'left',
                            title: {
                                display: true,
                                text: 'Bunch Weight (kg)',
                                color: '#34d399',
                                font: {
                                    weight: 'bold'
                                }
                            },
                            grid: {
                                color: 'rgba(255, 255, 255, 0.05)'
                            },
                            ticks: {
                                color: '#9ca3af'
                            },
                            min: 0,
                            max: 40
                        },
                        yRate: {
                            type: 'linear',
                            position: 'right',
                            title: {
                                display: true,
                                text: 'Growth Rate (kg/wk)',
                                color: '#a3e635',
                                font: {
                                    weight: 'bold'
                                }
                            },
                            grid: {
                                drawOnChartArea: false // Avoid clutter grid overlapping
                            },
                            ticks: {
                                color: '#9ca3af'
                            },
                            min: 0,
                            max: 6
                        }
                    }
                },
                // Draw dynamic vertical selection cursor line plugin
                plugins: [{
                    id: 'verticalLine',
                    afterDatasetsDraw: (chart) => {
                        if (selectedWeek !== null) {
                            const activeIndex = chart.data.labels.indexOf(parseFloat(selectedWeek.toFixed(1)));
                            if (activeIndex !== -1) {
                                const ctx = chart.ctx;
                                const meta = chart.getDatasetMeta(0);
                                if (meta.data[activeIndex]) {
                                    const x = meta.data[activeIndex].x;
                                    const topY = chart.scales.yWeight.top;
                                    const bottomY = chart.scales.yWeight.bottom;

                                    // Draw selection line
                                    ctx.save();
                                    ctx.beginPath();
                                    ctx.moveTo(x, topY);
                                    ctx.lineTo(x, bottomY);
                                    ctx.lineWidth = 2;
                                    ctx.strokeStyle = 'rgba(255, 255, 255, 0.25)';
                                    ctx.setLineDash([4, 4]);
                                    ctx.stroke();

                                    // Draw intersection circles on plots
                                    const meta2 = chart.getDatasetMeta(1);
                                    const y1 = meta.data[activeIndex].y;
                                    const y2 = meta2.data[activeIndex].y;

                                    // Weight circle
                                    ctx.beginPath();
                                    ctx.arc(x, y1, 6, 0, 2 * Math.PI);
                                    ctx.fillStyle = '#34d399';
                                    ctx.fill();
                                    ctx.stroke();

                                    // Rate circle
                                    ctx.beginPath();
                                    ctx.arc(x, y2, 6, 0, 2 * Math.PI);
                                    ctx.fillStyle = '#a3e635';
                                    ctx.fill();
                                    ctx.stroke();

                                    ctx.restore();
                                }
                            }
                        }
                    }
                }]
            });
        }

        // --- STREAMING_CHUNK: Managing slider and parameter change listeners ---
        function refreshChartData() {
            const dataSets = generateData();
            chart.data.labels = dataSets.labels;
            chart.data.datasets[0].data = dataSets.weight;
            chart.data.datasets[1].data = dataSets.rate;
            
            // Auto scale Y axes depending on Carrying Capacity to keep layout neat
            chart.options.scales.yWeight.max = Math.ceil(L + 5);
            chart.options.scales.yRate.max = Math.ceil((k * L / 4) + 1);

            chart.update();
            updateUI();
        }

        // Event listener bindings
        window.onload = function() {
            initChart();

            // Set dynamic slider value monitors
            const sliderL = document.getElementById('slider-L');
            const sliderk = document.getElementById('slider-k');
            const sliderT0 = document.getElementById('slider-t0');
            const scrubberT = document.getElementById('scrubber-t');

            sliderL.addEventListener('input', (e) => {
                L = parseFloat(e.target.value);
                document.getElementById('val-L').innerText = `${L.toFixed(1)} kg`;
                refreshChartData();
            });

            sliderk.addEventListener('input', (e) => {
                k = parseFloat(e.target.value);
                document.getElementById('val-k').innerText = k.toFixed(2);
                refreshChartData();
            });

            sliderT0.addEventListener('input', (e) => {
                t0 = parseFloat(e.target.value);
                document.getElementById('val-t0').innerText = `${t0.toFixed(1)} wks`;
                refreshChartData();
            });

            scrubberT.addEventListener('input', (e) => {
                selectedWeek = parseFloat(e.target.value);
                updateUI();
            });

            // Initial UI Sync
            updateUI();
        };
    </script>
